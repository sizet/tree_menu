使用方式

01. 網頁內設定載入相關的檔案.
<script type="text/javascript" src="script/tm.js"></script>
<link rel="stylesheet" href="style/tm.css">


02. 網頁的 body 內定義一個 div 元素, 選單會放在此元素內, 元素的 id 自定.
<div id="container"></div>


03. 網頁的 head 內定義 script 區塊, 撰寫產生選單的程式.
<script type="text/javascript">
// 宣告紀錄選單資料的變數 (必須是全域變數).
var tm;

// 產生選單.
function body_load()
{
    // 產生選單物件.
    // 參數一 : 紀錄選單的變數的名稱.
    // 參數二 : 放置選單的 div 元素的 id.
    // 參數三 : 點擊選單後網頁要顯示在哪個 frame.
    tm = new treemenu("tm", "container", window.parent.arena);

    // 初始化.
    tm.init();


    // 增加選單的節點.
    // 參數一 : 節點的名稱, 不可重複.
    // 參數二 : 此節點要放在哪個父節點之下, 空字串表是此節點是根節點.
    // 參數三 : 此節點顯示的選單文字描述, 可以在文字中加入 <br> 換行.
    // 參數四 : 此節點要連結到哪個網頁, 可以是空字串.
    tm.add("device", "", "Network Device", "page/device.html");

    // device -> descript.
    tm.add("descript", "device", "Descript", "page/descript.html");


    // device -> system.
    tm.add("system", "device", "System<br>Setting", "page/system.html");

    // device -> system -> date_time.
    tm.add("date_time", "system", "Date/Time", "page/date_time.html");


    // device -> wifi.
    tm.add("wifi", "device", "Wi-Fi", "page/wifi.html");

    // device -> wifi -> radio.
    tm.add("radio", "wifi", "Radio Setting", "page/radio.html");

    // device -> wifi -> wps.
    tm.add("wps", "wifi", "Wi-Fi<br>Protected<br>Setup", "page/wps.html");


    // device -> wifi -> limit.
    tm.add("limit", "wifi", "Limit", "page/limit.html");

    // device -> wifi -> limit -> station.
    tm.add("station", "limit", "Station", "page/station.html");

    // device -> wifi -> limit -> schedule.
    tm.add("schedule", "limit", "Schedule", "page/schedule.html");


    // device -> log.
    tm.add("log", "device", "Log", "");

    // device -> log -> security.
    tm.add("security", "log", "Security", "");

    // device -> log -> security -> attack.
    tm.add("attack", "security", "Attack", "page/attack.html");


    // 選單的節點增加完畢, 顯示選單.
    tm.show();
}
</script>


04. 網頁的 body 設定載入後執行產生選單的函式.
<body onload="body_load">


其他-01. tm.js 內紀錄圖片路徑的變數需要指定正確.
// 此路徑必須以使用的網頁為基準.
var image_path = "../source/image/";


其他-02. ${treemenu}.do_trigger("${name}") 此函式可以直接定位到指定的選單節點.
// 定位到 wifi -> limit -> station 此節點.
tm.do_trigger("station");


其他-03. 如何清除目前的選單並重新產生.
// 先初始化.
tm.init();

// 增加選單的節點.
tm.add(...);
...

// 顯示選單.
tm.show();
